Analysis of Bitcoin in R

Downloading Bitcoin data in R

First thing we need is data. In R, the quantmod package provides an API to pull financial data from Yahoo, Quandl, FRED (Federal Reserve Economic Database), google, oanda and MySQL.

btc <- getSymbols("BTCUSD", src = "alphavantage", api.key = "LMBDGRVLCXL8NHHP",
           output.size = "full", auto.assign = F)

BTC Price Chart

highchart(type = "stock") %>%
  hc_title(text = "Bitcoin Price Chart") %>%
  hc_add_series(btc[, 4], name = "BTC") %>%
  hc_add_theme(font_theme) %>%
  hc_navigator(enabled = FALSE) %>%
  hc_scrollbar(enabled = FALSE) %>%
  hc_exporting(enabled = TRUE, scale = 4) %>%
  hc_legend(enabled = TRUE) %>%
  hc_colors(c("blue", "orange", "green", "deeppink")) %>% 
  hc_credits(enabled = TRUE, text = "Source: Alpha Vantage")

BTC Daily Returns

btc_ret <- dailyReturn(btc[,4])

highchart(type = "stock") %>%
  hc_title(text = "Bitcoin Daily Returns") %>%
  hc_add_series(btc_ret["2017-10/2020"], name = "BTC") %>%
  hc_add_theme(font_theme) %>% 
  hc_navigator(enabled = FALSE) %>%
  hc_scrollbar(enabled = FALSE) %>%
  hc_exporting(enabled = TRUE, scale = 4) %>%
  hc_legend(enabled = TRUE) %>%
  hc_colors(c("#12326b")) %>% 
  hc_credits(enabled = TRUE, text = "Authors own calculations, data from Alpha Vantage")

Bitcoin Volatility

sd_roll <- rollify(sd, window = 365)

btc_vol <- sd_roll(btc_ret) * sqrt(365)

btc_vol <- xts(btc_vol, order.by = index(btc_ret))

highchart(type = "stock") %>%
  hc_title(text = "Bitcoin Annualised Daily Volatility") %>%
  hc_add_series(btc_vol["2015-10/2020"], name = "BTC") %>%
  hc_add_theme(font_theme) %>%
  hc_navigator(enabled = FALSE) %>%
  hc_scrollbar(enabled = FALSE) %>%
  hc_exporting(enabled = TRUE, scale = 4) %>%
  hc_legend(enabled = TRUE) %>%
  hc_colors(c("#12326b")) %>% 
  hc_yAxis(labels = list(
    formatter = JS('function() {
      return this.value*100+"%";
    }')
  )) %>%
  hc_tooltip(formatter = JS("function () {
      return '' + this.points[0].series.name + ': <b>' + Math.round(this.y * 100) +
        '</b>%';
    }"))

BTC Cumulative Return

cumret <- cumprod(1+btc_ret)

highchart(type = "stock") %>%

  hc_add_theme(font_theme) %>%

  hc_scrollbar(enabled = FALSE) %>%
  hc_exporting(enabled = TRUE, scale = 4) %>%
  hc_legend(enabled = TRUE) %>%
